<reasoning_guidance>
Progress tracking uses SSE (Server-Sent Events) instead of WebSocket because:

**SDK Integration:**
- Claude Agent SDK PostToolUse hooks emit progress events during pipeline execution
- Events flow: SDK tool → PostToolUse hook → SSE callback registry → EventSource
- This matches the backend architecture in 06-04 and 06-05

**SSE Advantages over WebSocket:**
- Simpler implementation (no handshake, automatic reconnection)
- Unidirectional (server-to-client only - perfect for progress updates)
- Built-in EventSource API in browsers
- Works over HTTP/2, better proxy/CDN support
- Matches SDK hooks pattern (emit-only, no bidirectional communication)

**Event Flow:**
1. StoryProgress page connects to `/api/sse/stories/{id}/progress?token=jwt`
2. Backend registers SSE callback with SDK hooks system
3. SDK executes pipeline, PostToolUse hooks emit events
4. Callbacks forward events to SSE stream
5. EventSource receives events, React state updates
</reasoning_guidance>

```tsx
export function Dashboard() {
  const [stories, setStories] = useState<Story[]>([]);

function StoryCard({ story, onDelete }: { story: Story; onDelete: (id: string) => void }) {
  const navigate = useNavigate();
  const repoName = story.repository_url.split("/").slice(-2).join("/");

  function handleClick() {
    if (story.status === "complete") {
      navigate(`/stories/${story.id}/player`);
    } else if (["pending", "analyzing", "generating", "synthesizing"].includes(story.status)) {
      navigate(`/stories/${story.id}/progress`);
    }
  }

  return (
    <div
      className="bg-slate-800 rounded-xl p-5 border border-slate-700 hover:border-slate-600 cursor-pointer transition-all"
      onClick={handleClick}
    >
      <div className="flex items-start justify-between">
        <div className="flex-1">
          <h3 className="text-lg font-medium text-white mb-1">
            {story.title || repoName}
          </h3>
          <p className="text-slate-400 text-sm mb-3">{repoName}</p>

          <div className="flex items-center gap-4 text-sm">
            <StatusBadge status={story.status} />
            <span className="text-slate-500 capitalize">{story.style}</span>
            {story.total_duration_seconds && (
              <span className="text-slate-500 flex items-center gap-1">
                <Clock className="h-4 w-4" />
                {Math.round(story.total_duration_seconds / 60)} min
              </span>
            )}
            <span className="text-slate-500">
              {formatDistanceToNow(new Date(story.created_at), { addSuffix: true })}
            </span>
          </div>
        </div>

        <div className="flex items-center gap-2" onClick={(e) => e.stopPropagation()}>
          {story.status === "complete" && (
            <Button
              size="sm"
              variant="ghost"
              className="text-indigo-400 hover:text-indigo-300"
              onClick={() => navigate(`/stories/${story.id}/player`)}
            >
              <Play className="h-4 w-4" />
            </Button>
          )}
          <Button
            size="sm"
            variant="ghost"
            className="text-slate-400 hover:text-red-400"
            onClick={() => onDelete(story.id)}
          >
            <Trash2 className="h-4 w-4" />
          </Button>
        </div>
      </div>
    </div>
  );
}
```

```typescript
/**
 * SSE client for Code Story progress events.
 *
 * Integrates with backend SSE router which receives events from
 * Claude Agent SDK PostToolUse hooks during pipeline execution.
 */

export interface SSEEvent {
  event: string;
  data: Record<string, unknown>;
}

export interface SSEClientOptions {
  onOpen?: () => void;
  onError?: (error: Event) => void;
  onMessage?: (event: SSEEvent) => void;
  onEvent?: Record<string, (data: Record<string, unknown>) => void>;
}

export function createSSEClient(
  url: string,
  options: SSEClientOptions = {}
): { close: () => void } {
  const eventSource = new EventSource(url);

  eventSource.onopen = () => {
    options.onOpen?.();
  };

  eventSource.onerror = (event) => {
    options.onError?.(event);
  };

  eventSource.onmessage = (event) => {
    try {
      const data = JSON.parse(event.data);
      options.onMessage?.({ event: "message", data });
    } catch (e) {
      console.error("Failed to parse SSE message:", e);
    }
  };

  // Register named event handlers
  if (options.onEvent) {
    for (const [eventName, handler] of Object.entries(options.onEvent)) {
      eventSource.addEventListener(eventName, (event: MessageEvent) => {
        try {
          const data = JSON.parse(event.data);
          handler(data);
        } catch (e) {
          console.error(`Failed to parse SSE event ${eventName}:`, e);
        }
      });
    }
  }

  return {
    close: () => eventSource.close(),
  };
}

export function getSSEUrl(storyId: string): string {
  const token = localStorage.getItem("access_token");
  const baseUrl = import.meta.env.VITE_API_URL || "";
  return `${baseUrl}/api/sse/stories/${storyId}/progress?token=${token}`;
}
```

Create SSE hook (src/hooks/useSSE.ts):

```typescript
import { useState, useEffect, useCallback } from "react";
import { createSSEClient, getSSEUrl, SSEEvent } from "@/lib/sse";

export interface ProgressData {
  story_id: string;
  status: string;
  progress_percent: number;
  current_step: string;
  timestamp?: string;
  audio_url?: string;
  error?: string;
}

export interface UseSSEOptions {
  onProgress?: (data: ProgressData) => void;
  onComplete?: (data: ProgressData) => void;
  onError?: (data: ProgressData) => void;
  autoConnect?: boolean;
}

export function useSSE(storyId: string | undefined, options: UseSSEOptions = {}) {
  const [isConnected, setIsConnected] = useState(false);
  const [progress, setProgress] = useState<ProgressData | null>(null);
  const [error, setError] = useState<string | null>(null);

  const connect = useCallback(() => {
    if (!storyId) return () => {};

    const url = getSSEUrl(storyId);

    const client = createSSEClient(url, {
      onOpen: () => {
        setIsConnected(true);
        setError(null);
      },
      onError: () => {
        setError("Connection lost. Attempting to reconnect...");
        setIsConnected(false);
      },
      onEvent: {
        connected: (data) => {
          console.log("SSE connected:", data);
        },
        progress: (data) => {
          const progressData = data as ProgressData;
          setProgress(progressData);
          options.onProgress?.(progressData);
        },
        complete: (data) => {
          const progressData = data as ProgressData;
          setProgress(progressData);
          options.onComplete?.(progressData);
        },
        error: (data) => {
          const progressData = data as ProgressData;
          setProgress(progressData);
          setError(progressData.error || "Unknown error");
          options.onError?.(progressData);
        },
        ping: () => {
          // Keepalive - no action needed
        },
      },
    });

    return () => client.close();
  }, [storyId, options]);

  useEffect(() => {
    if (options.autoConnect !== false) {
      return connect();
    }
  }, [connect, options.autoConnect]);

  return {
    isConnected,
    progress,
    error,
    connect,
  };
}
```

```tsx
import { useState, useEffect } from "react";
import { useParams, useNavigate } from "react-router-dom";
import { Loader2, CheckCircle, AlertCircle, Code, FileText, Mic, ArrowLeft } from "lucide-react";
import { Button } from "@/components/ui/button";
import { useSSE, ProgressData } from "@/hooks/useSSE";

const STEPS = [
  { key: "analyzing", label: "Analyzing Repository", icon: Code },
  { key: "generating", label: "Generating Script", icon: FileText },
  { key: "synthesizing", label: "Creating Audio", icon: Mic },
];

export function StoryProgress() {
  const { storyId } = useParams<{ storyId: string }>();
  const navigate = useNavigate();
  const [localError, setLocalError] = useState<string | null>(null);

  const { isConnected, progress, error } = useSSE(storyId, {
    onComplete: (data) => {
      // Redirect to player after brief delay
      setTimeout(() => {
        navigate(`/stories/${storyId}/player`);
      }, 1500);
    },
    onError: (data) => {
      setLocalError(data.error || "Story generation failed");
    },
  });

  const displayProgress = progress || {
    status: "pending",
    progress_percent: 0,
    current_step: "Connecting...",
  };

  const currentStepIndex = STEPS.findIndex((s) => s.key === displayProgress.status);
  const displayError = localError || error;

  return (
    <div className="min-h-screen bg-slate-900 flex items-center justify-center p-4">
      <div className="max-w-lg w-full">
        {displayError ? (
          <div className="text-center">
            <AlertCircle className="h-16 w-16 text-red-400 mx-auto mb-4" />
            <h1 className="text-2xl font-bold text-white mb-2">Generation Failed</h1>
            <p className="text-slate-400 mb-6">{displayError}</p>
            <Button onClick={() => navigate("/dashboard")} variant="outline">
              <ArrowLeft className="h-4 w-4 mr-2" />
              Back to Dashboard
            </Button>
          </div>
        ) : displayProgress.status === "complete" ? (
          <div className="text-center">
            <CheckCircle className="h-16 w-16 text-green-400 mx-auto mb-4" />
            <h1 className="text-2xl font-bold text-white mb-2">Story Complete!</h1>
            <p className="text-slate-400 mb-6">Redirecting to player...</p>
          </div>
        ) : (
          <div className="bg-slate-800 rounded-xl p-8 border border-slate-700">
            <h1 className="text-2xl font-bold text-white mb-2 text-center">
              Creating Your Story
            </h1>
            <p className="text-slate-400 text-center mb-8">
              This usually takes 2-3 minutes
            </p>

            {/* Connection Status */}
            {!isConnected && (
              <div className="text-center text-yellow-400 text-sm mb-4">
                Connecting to progress stream...
              </div>
            )}

            {/* Progress Bar */}
            <div className="mb-8">
              <div className="flex justify-between text-sm text-slate-400 mb-2">
                <span>{displayProgress.current_step}</span>
                <span>{displayProgress.progress_percent}%</span>
              </div>
              <div className="h-2 bg-slate-700 rounded-full overflow-hidden">
                <div
                  className="h-full bg-indigo-500 transition-all duration-500"
                  style={{ width: `${displayProgress.progress_percent}%` }}
                />
              </div>
            </div>

            {/* Steps */}
            <div className="space-y-4">
              {STEPS.map((step, index) => {
                const Icon = step.icon;
                const isActive = index === currentStepIndex;
                const isComplete = index < currentStepIndex;

                return (
                  <div
                    key={step.key}
                    className={`flex items-center gap-4 p-4 rounded-lg ${
                      isActive ? "bg-indigo-500/10 border border-indigo-500/50" : "bg-slate-700/30"

                      ${isComplete ? "bg-green-500" : isActive ? "bg-indigo-500" : "bg-slate-600"}
                    `}>

                        <CheckCircle className="h-5 w-5 text-white" />
                      ) : isActive ? (
                        <Loader2 className="h-5 w-5 text-white animate-spin" />
                      ) : (
                        <Icon className="h-5 w-5 text-slate-400" />
                      )}
                    </div>
                    <span className={`font-medium ${
                      isActive ? "text-white" : isComplete ? "text-slate-300" : "text-slate-500"
                    }`}>
                      {step.label}
                    </span>
                  </div>
                );
              })}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
```
